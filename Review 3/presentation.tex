% --- PROPOSED FRAMEWORK: FPDAF WHAT/HOW/NOVELTY (3) ---
\begin{frame}{Proposed Framework: FPDAF}
    \begin{columns}[t]
        \begin{column}{0.32\textwidth}
            \begin{block}{WHAT is FPDAF?}
                \scriptsize
                \textbf{Federated Personalized Drift-Aware Attention Framework:}
                \begin{itemize}
                    \item A privacy-preserving Clinical Decision Support System (CDSS) for Sepsis early warning.
                    \item Collaborative training across hospital ICU nodes without centralizing raw records (HIPAA/DPDPA compliant).
                \end{itemize}
            \end{block}
        \end{column}
        
        \begin{column}{0.34\textwidth}
            \begin{block}{HOW does it work?}
                \scriptsize
                \begin{itemize}
                    \item \textbf{Sequence Modeling:} Extracts temporal clinical features using a shared Bidirectional LSTM.
                    \item \textbf{Ditto Personalization:} Fine-tunes localized classifier heads ($v_k$) to fit hospital demographics.
                    \item \textbf{Online CUSUM Trigger:} Audits prediction loss residuals locally to detect population drift.
                \end{itemize}
            \end{block}
        \end{column}
        
        \begin{column}{0.32\textwidth}
            \begin{alertblock}{OUR NOVELTY}
                \scriptsize
                \begin{itemize}
                    \item \textbf{Multi-Head Attention:} Integrates 4-head query-key self-attention to output explainable bedside timelines.
                    \item \textbf{CSSP Protocol:} Freezes backbones dynamically upon drift triggers, saving \textbf{38\% network bandwidth} and cutting node adaptation to \textbf{6 minutes}.
                \end{itemize}
            \end{alertblock}
        \end{column}
    \end{columns}
\end{frame}

% --- SYSTEM DESIGN & SPLIT WEIGHT FORMULAS (4) ---
\begin{frame}{Unified System Design \& Neural Formulations}
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \centering
            \includegraphics[width=\textwidth]{images/ArcDigram.jpeg}
            \vspace{0.1cm}
            \scriptsize \textbf{Figure 1:} FPDAF Split-Weight Model Architecture.
        \end{column}
        \begin{column}{0.52\textwidth}
            \begin{block}{Decoupled bi-level Optimization Model}
                \scriptsize
                \[ \hat{y}_t^k = h_{\phi^k} \left( \sum_{i=1}^{t} \alpha_i \cdot g_{\theta}(X_{i}^k) \right) \]
                \begin{itemize}
                    \item $g_{\theta}$: Shared global LSTM backbone ($\theta$).
                    \item $h_{\phi^k}$: Local private classifier head ($\phi^k$).
                \end{itemize}
            \end{block}
            
            \begin{block}{Temporal Multi-Head Self-Attention}
                \scriptsize
                \[ e_i = v_a^T \tanh(W_s s_i + b_a) \]
                \[ \alpha_i = \frac{\exp(e_i)}{\sum_{j=1}^t \exp(e_j)} \]
                Attributes risk probabilities to specific ICU sequence hours.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% --- DRIFT & CUSUM NEURAL TRIGGER (4) ---
\begin{frame}{Proactive Drift Adaptation: CUSUM Neural Trigger}
    \begin{columns}
        \begin{column}{0.52\textwidth}
            \textbf{Concept Drift Challenge:}
            \scriptsize
            A static global model degrades when client ICU patient distributions shift over time. 
            Standard Personalized FL (e.g. Ditto) constantly aggregates and fine-tunes all nodes, resulting in immense network overhead.
            
            \vspace{0.2cm}
            \textbf{The CUSUM Residual Monitor Trigger:}
            \begin{itemize}
                \item We monitor cumulative validation loss residuals ($e_r$) over rounds:
                \[ e_r = \mathcal{L}_{val, r} - \mu_0 \]
                \[ S_r = \max(0, S_{r-1} + e_r - \kappa) \]
                \item If $S_r > H$ (threshold $H = 3.0$), we declare institutional drift.
                \item Triggers \textbf{Client-Side Selective Personalization (CSSP)}: freezes global backbone layers $\theta$ and adapts classification heads $\phi^k$ locally.
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{Bandwidth Savings Benefits}
                \scriptsize
                \begin{itemize}
                    \item Aggregation is bypassed during stable performance rounds.
                    \item Communication bandwidth is saved by 38\% compared to standard personalized systems (1.52 GB vs 2.48 GB).
                    \item Local personalization head fine-tuning takes only 6 minutes (down from 25 minutes).
                \end{itemize}
            \end{block}
            \centering
            \includegraphics[width=0.9\textwidth]{images/Dataflow2.png}
        \end{column}
    \end{columns}
\end{frame}
